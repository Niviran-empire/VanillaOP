			planet_crime_add = -10
		}
	}

	weight = {

		modifier = {
			mult = value:job_weights_modifier|JOB|enforcer|RESOURCE|unity|FACTOR|0.25|

		}

		modifier = {
			factor = 10
			has_job = enforcer
		}
		modifier = {
			factor = 10

			NOT = { has_job = enforcer }

			factor = 0.01
			planet = { planet_crime <= 15}
			NOT = { has_job = enforcer }
		}

		modifier = {
			factor = 0.2

		modifier = {

		}
		modifier = {
			factor = 1.5

		}
