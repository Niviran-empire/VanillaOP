select_suitable_spawn_system = {
	if = {
		limit = { is_nomadic = no }
		random_system_within_border = {
			limit = {
				is_capital_system = no
				NOT = {
					any_neighbor_system = {
						OR = {
							NOR = {
								exists = owner
								has_star_flag = hidden_treasure_system
							}
							AND = {
								exists = owner
								owner = { NOT = { is_same_value = root } }
							}
						}
					}
				}
			}
			weights = {
				base = 1
				modifier = {
					factor = 0.01
					OR = {
						has_star_flag = treasure_hunters_spawn_pinpoint
						has_star_flag = hidden_treasure_system
					}
				}
			}
			set_star_flag = treasure_hunters_spawn_pinpoint
			save_event_target_as = suitable_spawn_system
		}
	}
	else = {
		# Nomad path: use waystations as spatial anchors instead of border systems
		random_owned_nonprimary_starbase = {
			limit = {
				is_waystation_starbase = yes
				solar_system = {
					NOT = {
						any_neighbor_system = {
							OR = {
								NOR = {
									exists = owner
									has_star_flag = hidden_treasure_system
								}
								AND = {
									exists = owner
									owner = { NOT = { is_same_value = root } }
								}
							}
						}
					}
				}
			}
			weights = {
				base = 1
				modifier = {
					factor = 0.01
					solar_system = {
						OR = {
							has_star_flag = treasure_hunters_spawn_pinpoint
							has_star_flag = hidden_treasure_system
						}
					}
				}
				# Favor waystations in active travel corridors
				modifier = {
					add = 2
					has_neighboring_waystation = yes
				}
				modifier = {
					add = 2
					solar_system = {
						count_neighbor_system = {
							count >= 2
							limit = {
								any_fleet_in_system = {
									is_waystation_ship = yes
									owner = { is_same_value = root }
								}
							}
						}
					}
				}
				# Favor waystations near owned Arkships
				modifier = {
					add = 3
					solar_system = {
						any_neighbor_system = {
							any_fleet_in_system = {
								is_arkship_fleet = yes
								owner = { is_same_value = root }
							}
						}
					}
				}
			}
			solar_system = {
				set_star_flag = treasure_hunters_spawn_pinpoint
				save_event_target_as = suitable_spawn_system
			}
		}
	}
	if = {
		limit = {
			NOT = { exists = event_target:suitable_spawn_system }
		}
		if = { #Fallback if you are nomadic we just grab a neighboring system to your arkship
			limit = { is_nomadic = yes }
			capital_scope.solar_system = {
				random_neighbor_system = {
					solar_system = {
						save_event_target_as = suitable_spawn_system
					}
				}
			}
		}
		else = {
			capital_scope.solar_system = {
				random_neighbor_system = {
					limit = {
						exists = owner
						owner = { is_same_value = root }
					}
					save_event_target_as = suitable_spawn_system
				}
			}
		}
	}
	# Select capital if there is no other solution (non-nomad only — nomads handled above)
	if = {
		limit = {
			NOT = { exists = event_target:suitable_spawn_system }
			is_nomadic = no
		}
		capital_scope.solar_system = {
			save_event_target_as = suitable_spawn_system
		}
	}
}
