# Root = country
# This = species
can_modify_species = {
	if = {
		limit = { root = { is_ai = yes } }
		root = { count_exact_species = { count < 3 limit = { OR = { is_same_species = root is_subspecies = root } } } }
	}
	else = { OR = {
		NOT = { has_trait = trait_self_modified }
		is_same_species = root
	} }
}
