namespace = pirate
# Pirates Sighted On Action
event = {
	id = pirate.4
	hide_window = yes
	is_triggered_only = yes

	trigger = { always = no }
	
	immediate = {
		every_playable_country = {
			limit = {
				NOT = { has_country_flag = birth_of_piracy }
				any_system_within_border = {
					has_trade_route = yes
					trade_intercepted_value > 0
				}
			}
			country_event = { id = pirate.5 }
		}
	}
}
