namespace = paragon_2

# On destroying enemy ship.
country_event = {#_|VOP新增|添加trigger
	id = paragon_2.1005
	hide_window = yes
	is_triggered_only = yes

	trigger = {#_|VOP新增|
		exists = fromfrom
		fromfrom? = { 
			exists = fleet.leader 
			fleet.leader = {
				OR = {
					has_trait = leader_trait_corsair
					has_trait = leader_trait_corsair_2
					has_trait = leader_trait_shipbreaker
					has_trait = leader_trait_shipbreaker_2
					has_trait = leader_trait_juryrigger
					has_trait = leader_trait_juryrigger_2
					has_trait_tier1or2 = { TRAIT = leader_trait_crusader }
				}
			}
		}
	}
	immediate = {
		# Cache the destroyed ship so trait rewards can scale off its used naval capacity.
		fromfromfrom? = { save_event_target_as = paragon_destroyed_ship }
		fromfrom? = { # Winning ship.
			if = {
				limit = { exists = fleet.leader }
				fleet.leader = {

					# Corsair / Corsair II
					if = {
						limit = {
							OR = {
								has_trait = leader_trait_corsair
								has_trait = leader_trait_corsair_2
							}
							ROOT.from? = { # Owner of losing ship.
								is_default_or_fallen = yes
							}
						}
						if = {
							limit = { has_trait = leader_trait_corsair }
							root? = { # Owner of winning ship.
								add_resource = {
									energy = @trait_corsair_amt
									mult = value:trait_kill_naval_cap_scaling|MULT|value:hells_heart_effect|
								}
							}
						}
						else = {
							root? = { # Owner of winning ship.
								add_resource = {
									energy = @trait_corsair_2_amt
									mult = value:trait_kill_naval_cap_scaling|MULT|value:hells_heart_effect|
								}
							}
						}
					}

					# Crusader / Crusader II
					if = {
						limit = {
							has_trait_tier1or2 = {
								TRAIT = leader_trait_crusader
							}
							ROOT.from? = { # Owner of losing ship.
								is_default_or_fallen = yes
							}
						}

						root? = {
							if = {
								limit = { prev? = { has_trait = leader_trait_crusader } }
								add_resource = {
									unity = @trait_crusader_fleet_amt
									mult = value:trait_kill_naval_cap_scaling|MULT|1|
								}
							}
							else = {
								add_resource = {
									unity = @trait_crusader_2_fleet_amt
									mult = value:trait_kill_naval_cap_scaling|MULT|1|
								}
							}
						}
					}

					# Shipbreaker / Shipbreaker II
					if = {
						limit = {
							OR = {
								has_trait = leader_trait_shipbreaker
								has_trait = leader_trait_shipbreaker_2
							}
							ROOT.from? = { # Owner of losing fleet.
								is_default_or_fallen = yes
							}
						}
						if = {
							limit = { has_trait = leader_trait_shipbreaker }
							root? = {
								add_resource = {
									alloys = @trait_shipbreaker_amt
									mult = value:trait_kill_naval_cap_scaling|MULT|value:hells_heart_effect|
								}
							}
						}
						else = {
							root? = {
								add_resource = {
									alloys = @trait_shipbreaker_2_amt
									mult = value:trait_kill_naval_cap_scaling|MULT|value:hells_heart_effect|
								}
							}
						}
					}

					# Juryrigger / Juryrigger II
					if = {
						limit = {
							OR = {
								has_trait = leader_trait_juryrigger
								has_trait = leader_trait_juryrigger_2
							}
						}
						ROOT.fromfrom? = {
							if = {
								limit = { prev? = { has_trait = leader_trait_juryrigger } }
								repair_percentage = @trait_juryrigger_amt
							}
							else = {
								repair_percentage = @trait_juryrigger_2_amt
							}
						}
					}

					# Plunderer / Plunderer II
					if = {
						limit = {
							OR = {
								has_trait = leader_trait_plunderer
								has_trait = leader_trait_plunderer_2
							}
						}
						ROOT.fromfrom? = {
							if = {
								limit = { prev? = { has_trait = leader_trait_plunderer } }
								root? = {
									random_list = {
										1 = {
											add_resource = {
												trade = @trait_plunderer_amt
												mult = value:trait_kill_naval_cap_scaling|MULT|1|
											}
										}
										1 = {
											add_resource = {
												consumer_goods = @trait_plunderer_amt
												mult = value:trait_kill_naval_cap_scaling|MULT|1|
											}
										}
									}
								}
							}
							else = {
								root? = {
									random_list = {
										1 = {
											add_resource = {
												trade = @trait_plunderer_2_amt
												mult = value:trait_kill_naval_cap_scaling|MULT|1|
											}
										}
										1 = {
											add_resource = {
												consumer_goods = @trait_plunderer_2_amt
												mult = value:trait_kill_naval_cap_scaling|MULT|1|
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}
}

	trigger = { str_paragon_2_1005_trigger_1 = yes }	#_|VOP新增|
